generator client {

datasource db {
  provider = "postgresql"
}

// ============================================
// AUTHENTICATION MODELS
// ============================================

model User {
  id               String      @id
  name             String
  email            String      @unique
  emailVerified    Boolean     @default(false)
  image            String?
  sessions         Session[]
  accounts         Account[]
  role             String      @default("user")
  phones           Phone[]
  userStores       UserStore[]
  profileCompleted Boolean     @default(false)
  createdAt        DateTime    @default(now())
  updatedAt        DateTime    @default(now()) @updatedAt
}

model Session {
  id        String   @id
  expiresAt DateTime
  token     String   @unique
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
  ipAddress String?
  userAgent String?
  userId    String

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)
}

model Account {
  id                    String    @id
  accountId             String
  providerId            String
  userId                String
  accessToken           String?
  refreshToken          String?
  idToken               String?
  accessTokenExpiresAt  DateTime?
  refreshTokenExpiresAt DateTime?
  scope                 String?
  password              String?
  createdAt             DateTime  @default(now())
  updatedAt             DateTime  @updatedAt

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)
}

model Verification {
  id         String   @id
  identifier String
  value      String
  expiresAt  DateTime
  createdAt  DateTime @default(now())
  updatedAt  DateTime @default(now()) @updatedAt
}

// ============================================
// BUSINESS MODELS
// ============================================

model Phone {
  id        String   @id @default(cuid())
  number    String
  isPrimary Boolean  @default(false)
  userId    String
  createdAt DateTime @default(now())
  updatedAt DateTime @default(now()) @updatedAt

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([userId, number])
  @@index([userId])
}

model Store {
  id         String      @id @default(cuid())
  name       String      @unique
  isActive   Boolean     @default(true)
  warehouses Warehouse[]
  salesAreas SalesArea[]
  outflows   Outflow[]
  userStores UserStore[]
  createdAt  DateTime    @default(now())
  updatedAt  DateTime    @default(now()) @updatedAt

  @@index([name])
}

model UserStore {
  id        String   @id @default(cuid())
  userId    String
  storeId   String
  isActive  Boolean  @default(true)
  createdAt DateTime @default(now())
  updatedAt DateTime @default(now()) @updatedAt

  user  User  @relation(fields: [userId], references: [id], onDelete: Cascade)
  store Store @relation(fields: [storeId], references: [id], onDelete: Cascade)

  // IMPORTANTE: Esto previene duplicados
  @@unique([userId, storeId])
  @@index([userId])
  @@index([storeId])
}

model Warehouse {
  id        String    @id @default(cuid())
  storeId   String
  name      String
  isActive  Boolean   @default(true)
  inflows   Inflow[]
  outflows  Outflow[]
  createdAt DateTime  @default(now())
  updatedAt DateTime  @default(now()) @updatedAt

  store Store @relation(fields: [storeId], references: [id], onDelete: Restrict)

  @@unique([name, storeId])
  @@index([storeId])
  @@index([name])
}

model SalesArea {
  id        String     @id @default(cuid())
  storeId   String
  name      String
  isActive  Boolean    @default(true)
  withdraws Withdraw[]
  outflows  Outflow[]
  sales     Sale[]
  createdAt DateTime   @default(now())
  updatedAt DateTime   @default(now()) @updatedAt

  store Store @relation(fields: [storeId], references: [id], onDelete: Restrict)

  @@unique([name, storeId])
  @@index([storeId])
  @@index([name])
}

model Code {
  id        String    @id
  name      String    @unique
  products  Product[]
  createdAt DateTime  @default(now())
  updatedAt DateTime  @default(now()) @updatedAt

  @@index([name])
}

model Product {
  id        String    @id
  codeId    String
  name      String    @unique
  costPrice Decimal   @db.Decimal(10, 2)
  salePrice Decimal   @db.Decimal(10, 2)
  unit      String
  inflows   Inflow[]
  outflows  Outflow[]
  sales     Sale[]
  createdAt DateTime  @default(now())
  updatedAt DateTime  @default(now()) @updatedAt

  code Code @relation(fields: [codeId], references: [id], onDelete: Restrict)

  @@index([name])
  @@index([codeId])
}

model Company {
  id         String   @id @default(cuid())
  name       String   @unique
  isProvider Boolean  @default(false)
  inflows    Inflow[]
  createdAt  DateTime @default(now())
  updatedAt  DateTime @default(now()) @updatedAt

  @@index([name])
}

// ============================================
// TRANSACTION MODELS
// ============================================

model Inflow {
  id          String   @id @default(cuid())
  warehouseId String
  type        String
  date        DateTime @default(now()) @db.Date
  providerId  String?
  payment     String
  inNumber    String
  serial      String
  productId   String
  quantity    Int      @default(0)
  amount      Decimal  @db.Decimal(10, 2)
  createdAt   DateTime @default(now())
  updatedAt   DateTime @default(now()) @updatedAt

  warehouse Warehouse @relation(fields: [warehouseId], references: [id], onDelete: Restrict)
  product   Product   @relation(fields: [productId], references: [id], onDelete: Restrict)
  provider  Company?  @relation(fields: [providerId], references: [id], onDelete: SetNull)

  @@index([warehouseId])
  @@index([productId])
  @@index([providerId])
  @@index([date])
}

model Withdraw {
  id        String   @id @default(cuid())
  areaId    String
  date      DateTime @default(now()) @db.Date
  amount    Decimal  @db.Decimal(10, 2)
  createdAt DateTime @default(now())
  updatedAt DateTime @default(now()) @updatedAt

  area SalesArea @relation(fields: [areaId], references: [id], onDelete: Restrict)

  @@index([areaId])
  @@index([date])
}

model Outflow {
  id          String   @id @default(cuid())
  warehouseId String
  type        String
  date        DateTime @default(now()) @db.Date
  endAreaId   String?
  endStoreId  String?
  outNumber   String
  productId   String
  quantity    Int      @default(0)
  amount      Decimal  @default(0) @db.Decimal(10, 2)
  createdAt   DateTime @default(now())
  updatedAt   DateTime @default(now()) @updatedAt

  warehouse Warehouse  @relation(fields: [warehouseId], references: [id], onDelete: Restrict)
  endArea   SalesArea? @relation(fields: [endAreaId], references: [id], onDelete: SetNull)
  endStore  Store?     @relation(fields: [endStoreId], references: [id], onDelete: SetNull)
  product   Product    @relation(fields: [productId], references: [id], onDelete: Restrict)

  @@index([warehouseId])
  @@index([productId])
  @@index([endAreaId])
  @@index([endStoreId])
  @@index([date])
}

model Sale {
  id        String   @id @default(cuid())
  areaId    String
  date      DateTime @default(now()) @db.Date
  payment   String
  productId String
  quantity  Int      @default(0)
  createdAt DateTime @default(now())
  updatedAt DateTime @default(now()) @updatedAt

  area    SalesArea @relation(fields: [areaId], references: [id], onDelete: Restrict)
  product Product   @relation(fields: [productId], references: [id], onDelete: Restrict)

  @@index([areaId])
  @@index([productId])
  @@index([date])
}
